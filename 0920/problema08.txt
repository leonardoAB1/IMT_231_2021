Problema 8
Determine la complejidad de los métodos intersect 
y difference que implementó en el problema 6.
Complejidad: O(n)
    def intersect(self, setB):                          #O(n)
        newSet=Set()                                    #O(1)
        for element in setB:                            #O(n)
            if element in self:                         #O(1)
                newSet._elements.append(element)        #O(1)
        return newSet                                   #O(1)
Complejidad: O(n)
    def difference(self, setB):                         #O(n)
        newSet=Set()                                    #O(1)
        for element in self._elements:                  #O(n)
            if element not in setB:                     #O(1)
                newSet._elements.append(element)        #O(1)
        for element in setB:                            #O(n)
            if element not in self:                     #O(1)
                newSet._elements.append(element)        #O(1)
        return newSet                                   #O(1)
