Problema 8
Determine la complejidad de los métodos intersect 
y difference que implementó en el problema 6.

    def intersect(self, setB):                          #O(n)
        newSet=Set()                                    #O(1)
        for element in setB:                            #O(n)*1
            if element in self:
                newSet._elements.append(element)        #O(1)
        return newSet
    
    def difference(self, setB):                         #O(n)
        newSet=Set()                                    #O(1)
        for element in setB:                            #O(n)
            if element not in self:                     
                newSet._elements.append(element)        #O(1)
        for element in self._elements:
            if element not in setB:
                newSet._elements.append(element)        #O(1)
        return newSet                                   #O(1)
